CREATE TABLE STG_GUEST_STAY_AUDIT (
    AUDIT_ID        NUMBER GENERATED ALWAYS AS IDENTITY,
    STAYID          NUMBER,
    ACTION_TYPE     VARCHAR2(10),
    OLD_STATUS      VARCHAR2(26),
    NEW_STATUS      VARCHAR2(26),
    OLD_TOTAL       NUMBER,
    NEW_TOTAL       NUMBER,
    CHANGED_BY      VARCHAR2(50),
    CHANGED_DATE    DATE
);


CREATE OR REPLACE TRIGGER TRG_STG_GUEST_STAY_AUDIT
AFTER INSERT OR UPDATE OR DELETE ON STG_GUEST_STAY
FOR EACH ROW
BEGIN
    IF INSERTING THEN
        INSERT INTO STG_GUEST_STAY_AUDIT (
            STAYID, ACTION_TYPE, NEW_STATUS, NEW_TOTAL, CHANGED_BY, CHANGED_DATE
        )
        VALUES (
            :NEW.STAYID, 'INSERT', :NEW.STATUS, :NEW.TOTALAMOUNT, USER, SYSDATE
        );

    ELSIF UPDATING THEN
        INSERT INTO STG_GUEST_STAY_AUDIT (
            STAYID, ACTION_TYPE, OLD_STATUS, NEW_STATUS,
            OLD_TOTAL, NEW_TOTAL, CHANGED_BY, CHANGED_DATE
        )
        VALUES (
            :OLD.STAYID, 'UPDATE',
            :OLD.STATUS, :NEW.STATUS,
            :OLD.TOTALAMOUNT, :NEW.TOTALAMOUNT,
            USER, SYSDATE
        );

    ELSIF DELETING THEN
        INSERT INTO STG_GUEST_STAY_AUDIT (
            STAYID, ACTION_TYPE, OLD_STATUS, OLD_TOTAL, CHANGED_BY, CHANGED_DATE
        )
        VALUES (
            :OLD.STAYID, 'DELETE', :OLD.STATUS, :OLD.TOTALAMOUNT, USER, SYSDATE
        );
    END IF;
END;


CREATE OR REPLACE PROCEDURE GET_DAILY_METRICS (
    P_OCCUPIED_ROOMS OUT NUMBER,
    P_TOTAL_REVENUE  OUT NUMBER
)
AS
BEGIN
    SELECT COUNT(DISTINCT ROOMID)
    INTO P_OCCUPIED_ROOMS
    FROM STG_GUEST_STAY
    WHERE TRUNC(CHECKINDATETIME) <= TRUNC(SYSDATE)
      AND TRUNC(
            TO_DATE(CHECKOUTDATETIME,'YYYY-MM-DD HH24:MI:SS')
          ) >= TRUNC(SYSDATE);
    SELECT NVL(SUM(TOTALAMOUNT),0)
    INTO P_TOTAL_REVENUE
    FROM STG_GUEST_STAY
    WHERE TRUNC(CHECKINDATETIME) = TRUNC(SYSDATE);

END;

SET SERVEROUTPUT ON;
DECLARE
    V_ROOMS NUMBER;
    V_REVENUE NUMBER;
BEGIN
    GET_DAILY_METRICS(V_ROOMS, V_REVENUE);

    DBMS_OUTPUT.PUT_LINE('Rooms Occupied: ' || V_ROOMS);
    DBMS_OUTPUT.PUT_LINE('Total Revenue: ' || V_REVENUE);
END;
